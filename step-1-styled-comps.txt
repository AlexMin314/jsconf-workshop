# Add a page

App.jsx - Set up the route
    1. import { StyledComp } from 'routes'
    2. <Route exact path={'/'} component={StyledComp} />

routes/StyledComp/StyledComp.jsx - Create a Button component

    // EXTERNAL DEPENDENCIES
    import React from 'react'

    const StyledComp = () => (
        return <button>Styled Component</button>
    )

    export default StyledComp

    Make it a Styled Component
    1. import styled from "styled-components"
    2. const CustomButton = styled.button`
                background: green
            `
    3. <CustomButton>Styled Component</CustomButton>
